local function hasWriteFile()
    return pcall(function() writefile("test.txt", "test") end)
end

local function getConfigPath(name)
    return Config.Folder .. "/" .. name .. ".json"
end

local function saveConfigToFile(name, data)
    if not hasWriteFile() then return false end
    local path = getConfigPath(name)
    local success, err = pcall(function()
        writefile(path, game:GetService("HttpService"):JSONEncode(data))
    end)
    if not success then warn("Save failed:", err) return false end
    return true
end

local function loadConfigFromFile(name)
    if not hasWriteFile() then return nil end
    local path = getConfigPath(name)
    local success, data = pcall(function()
        return readfile(path)
    end)
    if not success then return nil end
    local decoded = game:GetService("HttpService"):JSONDecode(data)
    return decoded
end

local function listConfigFiles()
    if not hasWriteFile() then return {} end
    local success, files = pcall(function()
        return listfiles(Config.Folder)
    end)
    if not success then return {} end
    local names = {}
    for _, file in ipairs(files) do
        local name = string.match(file, "([^/\\]+)%.json$")
        if name then table.insert(names, name) end
    end
    return names
end

local function deleteConfigFile(name)
    if not hasWriteFile() then return false end
    local path = getConfigPath(name)
    local success = pcall(function()
        delfile(path)
    end)
    return success
end

local function exportToClipboard()
    local data = { Version = "10.0", Name = Config.CurrentName, Date = os.date("%Y-%m-%d %H:%M:%S"), Settings = Settings }
    local json = game:GetService("HttpService"):JSONEncode(data)
    if setclipboard then setclipboard(json); return true end
    return false
end

local function importFromClipboard()
    if not getclipboard then return nil end
    local json = getclipboard()
    if not json or json == "" then return nil end
    local decoded = game:GetService("HttpService"):JSONDecode(json)
    return decoded
end

local function saveLastConfig(name)
    if hasWriteFile() then pcall(function() writefile(Config.LastFile, name) end) end
end

local function loadLastConfig()
    if hasWriteFile() then
        local success, name = pcall(function() return readfile(Config.LastFile) end)
        if success and name and name ~= "" then return name end
    end
    return nil
end

local function CreateSlider(parent, label, settingKey, minVal, maxVal, default, y, step)
    step = step or 1
    Settings[settingKey] = default
    local frame = Instance.new("Frame")
    frame.Parent = parent
    frame.BackgroundColor3 = Color3.fromRGB(40, 40, 47)
    frame.Position = UDim2.new(0, 0, 0, y)
    frame.Size = UDim2.new(1, 0, 0, 52)
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 6)

    local lbl = Instance.new("TextLabel")
    lbl.Parent = frame
    lbl.BackgroundTransparency = 1
    lbl.Size = UDim2.new(0.75, 0, 0, 22)
    lbl.Position = UDim2.new(0, 10, 0, 2)
    lbl.Font = Enum.Font.Gotham
    lbl.Text = label .. " : " .. tostring(default)
    lbl.TextColor3 = Color3.fromRGB(230, 230, 230)
    lbl.TextSize = 13
    lbl.TextXAlignment = Enum.TextXAlignment.Left

    local bg = Instance.new("Frame")
    bg.Parent = frame
    bg.BackgroundColor3 = Color3.fromRGB(60, 60, 70)
    bg.Position = UDim2.new(0, 10, 0, 28)
    bg.Size = UDim2.new(1, -20, 0, 6)
    Instance.new("UICorner", bg).CornerRadius = UDim.new(0, 3)

    local fill = Instance.new("Frame")
    fill.Parent = bg
    fill.BackgroundColor3 = Color3.fromRGB(0, 200, 255)
    fill.Size = UDim2.new((default - minVal) / (maxVal - minVal), 0, 1, 0)
    Instance.new("UICorner", fill).CornerRadius = UDim.new(0, 3)

    local dragging = false
    local function update(input)
        local rel = math.clamp((input.Position.X - bg.AbsolutePosition.X) / bg.AbsoluteSize.X, 0, 1)
        local val = math.round((minVal + (maxVal - minVal) * rel) / step) * step
        val = math.clamp(val, minVal, maxVal)
        Settings[settingKey] = val
        fill.Size = UDim2.new((val - minVal) / (maxVal - minVal), 0, 1, 0)
        lbl.Text = label .. " : " .. tostring(val)
    end

    bg.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true; update(input) end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then update(input) end
    end)

    widgetRefs.sliders[settingKey] = { fill = fill, label = lbl, labelText = label, min = minVal, max = maxVal }
    return frame
end

    btn.Position = UDim2.new(0.55, 0, 0.5, -14)

    btn.Font = Enum.Font.Gotham
    btn.Text = default
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.TextSize = 12
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 4)

    local isOpen = false; local list = nil
    btn.MouseButton1Click:Connect(function()
        if isOpen then if list then list:Destroy() end isOpen = false return end
        isOpen = true
        list = Instance.new("Frame")
        list.Parent = frame
        list.BackgroundColor3 = Color3.fromRGB(50, 50, 60)
        list.Position = UDim2.new(0.55, 0, 1, 2)
        list.Size = UDim2.new(0.4, -10, 0, #options * 24)
        Instance.new("UICorner", list).CornerRadius = UDim.new(0, 4)
        for i, opt in ipairs(options) do
            local b = Instance.new("TextButton")
            b.Parent = list
            b.BackgroundColor3 = (opt == Settings[settingKey]) and Color3.fromRGB(0, 180, 255) or Color3.fromRGB(50, 50, 60)
            b.Size = UDim2.new(1, 0, 0, 24)

                if callback then callback() end
            end)
        end
    end)

    return frame

    if showJitter then

        jitterAngleSlider.Visible = true
        jitterSpeedSlider.Visible = (jMode == "Sway")
        local lbl = jitterAngleSlider:FindFirstChildOfClass("TextLabel")

            if jMode == "Random" then lbl.Text = "Max Angle : " .. tostring(Settings.JitterAngle)
            else lbl.Text = "Amplitude : " .. tostring(Settings.JitterAngle) end
        end
    else
        jitterAngleSlider.Visible = false; jitterSpeedSlider.Visible = false
    end
end

RunService.RenderStepped:Connect(function(dt)
    -- Обновление HUD (Пинг вырезан, осталось 3 плашки)
    if Settings.HudEnabled then
        hudContainer.Visible = true
        
        -- FPS (счётчик кадров в секунду)
        fpsCounter = fpsCounter + 1
        if tick() - fpsTimer >= 1 then
            lblFps.Text = "FPS: " .. tostring(fpsCounter)
            fpsCounter = 0
            fpsTimer = tick()
        end
    else
        hudContainer.Visible = false
    end

    if tick() - lastESPUpdate > 0.5 then lastESPUpdate = tick(); updateESP() end

    local char = LocalPlayer.Character
    if not char or not char.Parent then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChild("Humanoid")
    if not root or not hum then return end

    if Settings.Noclip then
        for _, part in ipairs(char:GetDescendants()) do
            if part:IsA("BasePart") and part.CanCollide then part.CanCollide = false end
        end
    end

    if Settings.Fly then
        if not flyVelocity then
            flyVelocity = Instance.new("BodyVelocity"); flyVelocity.MaxForce = Vector3.new(4000, 4000, 4000); flyVelocity.Parent = root
            flyGyro = Instance.new("BodyGyro"); flyGyro.MaxTorque = Vector3.new(4000, 4000, 4000); flyGyro.Parent = root
        end
        local cam = Camera
        local move = Vector3.new(0, 0, 0)
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then move = move + cam.CFrame.LookVector * Vector3.new(1, 0, 1) end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then move = move - cam.CFrame.LookVector * Vector3.new(1, 0, 1) end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then move = move - cam.CFrame.RightVector * Vector3.new(1, 0, 1) end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then move = move + cam.CFrame.RightVector * Vector3.new(1, 0, 1) end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then move = move + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then move = move - Vector3.new(0, 1, 0) end
        if move.Magnitude > 0 then move = move.Unit * Settings.FlySpeed end
        flyVelocity.Velocity = move; flyGyro.CFrame = cam.CFrame
    else
        if flyVelocity then flyVelocity:Destroy(); flyVelocity = nil end
        if flyGyro then flyGyro:Destroy(); flyGyro = nil end
    end

    if Settings.AntiAim ~= "Off" and root then
        if Settings.AntiAim == "Spin" then
            root.CFrame = root.CFrame * CFrame.Angles(0, math.rad(Settings.SpinSpeed), 0)
        elseif Settings.AntiAim == "Jitter" then
            if Settings.JitterMode == "Random" then
                local rx = math.rad(math.random(-Settings.JitterAngle, Settings.JitterAngle))
                local ry = math.rad(math.random(-Settings.JitterAngle, Settings.JitterAngle))
                root.CFrame = root.CFrame * CFrame.Angles(rx, ry, 0)
            elseif Settings.JitterMode == "Sway" then
                jitterAngle = jitterAngle + math.rad(Settings.JitterSpeed)
                local angle = math.sin(jitterAngle) * math.rad(Settings.JitterAngle)
                root.CFrame = root.CFrame * CFrame.Angles(0, angle, 0)
            end
        end
    end

    if Settings.FakeLag then
        tickCounter = tickCounter + 1
        if tickCounter >= Settings.FakeLagTicks then
            tickCounter = 0
            local mode = Settings.FakeLagMode
            if mode == "Classic" then
                local back = root.CFrame.LookVector * -Settings.FakeLagAmplitude
                root.CFrame = root.CFrame + back
            elseif mode == "Choke" then
                hum.PlatformStand = true; root.Velocity = Vector3.new(0, 0, 0)
                spawn(function() wait(0.05); if hum and hum.Parent then hum.PlatformStand = false end end)
            elseif mode == "Teleport" then
                local dir = root.CFrame.LookVector
                if fakeLagState == 0 then root.CFrame = root.CFrame + dir * Settings.FakeLagAmplitude; fakeLagState = 1
                else root.CFrame = root.CFrame - dir * Settings.FakeLagAmplitude; fakeLagState = 0 end
            elseif mode == "PacketLoss" then
                if math.random() < 0.3 then
                    local back = root.CFrame.LookVector * -Settings.FakeLagAmplitude
                    root.CFrame = root.CFrame + back
                end
            elseif mode == "Jitter" then
                local offset = Vector3.new(math.random(-100, 100) / 100 * Settings.FakeLagAmplitude, 0, math.random(-100, 100) / 100 * Settings.FakeLagAmplitude)
                root.CFrame = root.CFrame + offset
            end
        end
    else
        tickCounter = 0; fakeLagState = 0
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local targetRoot = player.Character:FindFirstChild("HumanoidRootPart")
            if targetRoot then
                if Settings.Hitbox then
                    targetRoot.Size = Vector3.new(Settings.HitboxSize, Settings.HitboxSize, Settings.HitboxSize)
                    targetRoot.Color = Color3.fromRGB(255, 0, 0); targetRoot.Material = Enum.Material.Neon; targetRoot.Transparency = 0.5; targetRoot.CanCollide = false
                else
                    targetRoot.Size = Vector3.new(2, 2, 1); targetRoot.Transparency = 1
                end
            end
        end
    end
end)
